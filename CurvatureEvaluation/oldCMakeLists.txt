cmake_minimum_required(VERSION 3.16.3)

project(CurvatureEvaluator)

SET(CMAKE_CXX_STANDARD 11)
SET(CMAKE_CXX_STANDARD_REQUIRED 1)
SET(CMAKE_INCLUDE_CURRENT_DIR ON)

FIND_PACKAGE(GTest REQUIRED)
FIND_PACKAGE(Eigen3 3.3 REQUIRED)

link_directories(${Eigen_INCLUDE_DIRS})
include_directories(${GTEST_INCLUDE_DIRS})
                    
add_library(${PROJECT_NAME} STATIC 
    CubicEquationSolver.hpp
    CubicEquationSolver.cpp
    BsplineToBezier.cpp
    BsplineToBezier.hpp
    SecondOrderCurvatureEvaluator.hpp
    SecondOrderCurvatureEvaluator.cpp)

set(TEST_NAME ${PROJECT_NAME}_UnitTest)

add_executable(${TEST_NAME}
    UnitTestCubicEquationSolver.cpp
    UnitTestBsplineToBezier.cpp
    UnitTestSecondOrderCurvatureEvaluator.cpp)

target_link_libraries(${TEST_NAME} ${GTEST_LIBRARIES}
    pthread ${GTEST_MAIN_LIBRARIES} ${PROJECT_NAME}
    ${Eigen_LIBRARIES})
